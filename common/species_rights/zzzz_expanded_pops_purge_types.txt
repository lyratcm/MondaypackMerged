purge_matrix = {
	token = purge_matrix
	pop_escape_chance = 0.20
	pop_decline_rate = 100 # 1 pop per month, overrides BASE_POP_DECLINE, will only work on purge types
	
	pop_modifier = {
		pop_happiness = -10
	}

	potential = {
		host_has_dlc = "Synthetic Dawn Story Pack"
		from = {
			OR = {
				has_authority = auth_machine_intelligence
				has_country_flag = synthetic_empire
			}
		}
	}

	allow = {
		hidden_trigger = { exists = from }
		custom_tooltip = {
			fail_text = CANNOT_LIVESTOCK_MATRIX_MACHINES
			NOR = {
				has_trait = trait_mechanical
				has_trait = trait_machine_unit
			}
		}
		custom_tooltip = {
			fail_text = PURGES_ALLOWED
			from = {
				OR = {
					has_ethic = ethic_gestalt_consciousness
					has_valid_civic = civic_fanatic_purifiers
					has_policy_flag = purge_allowed
				}
				NOR = {
					OR = {
						has_valid_civic = civic_harm_hive
						has_valid_civic = civic_harm_machine
					}
				}
			}
		}
	}

	ai_will_do = {
		factor = 50
		modifier = {
			factor = 0
			from = { NOT = { has_valid_civic = "civic_machine_terminator" } }
			from = { NOT = { has_ethic = ethic_fanatic_xenophobe } }
		}		
		modifier = {
			factor = 0
			from = {
				has_valid_ai_personality = yes
				NOT = {
					has_ai_personality_behaviour = purger
				}
			}
		}		
	}
}


purge_processing = {
	token = purge_processing
	pop_escape_chance = 0.20
	pop_decline_rate = 100 # 1 pop per month, overrides BASE_POP_DECLINE, will only work on purge types
	
	pop_modifier = {
		pop_happiness = -10
	}

	potential = {
		host_has_dlc = "Utopia"
		exists = from
		from = {
			NOR = {
				AND = {
					has_authority = auth_machine_intelligence
					NOT = { has_valid_civic = civic_machine_consciousness }
				}
				has_country_flag = synthetic_empire
			}
		}
	}
	
	allow = {
		hidden_trigger = { exists = from }
		custom_tooltip = {
			fail_text = CAN_NOT_EAT_MACHINES
			OR = {
				NOR = {
					has_trait = trait_mechanical
					has_trait = trait_machine_unit
				}
				from = { has_valid_civic = civic_hive_devouring_swarm }
				# devouring swarms can CRONCH. in edge cases.
			}
		}
		custom_tooltip = {
			fail_text = "lonely_hive_mind_pops_kill_themselves_purge_type"
			is_lonely_hive_mind_pop = no
		}
		custom_tooltip = {
			fail_text = PURGES_ALLOWED
			from = {
				OR = {
					has_ethic = ethic_gestalt_consciousness
					has_valid_civic = civic_fanatic_purifiers
					has_policy_flag = purge_allowed
					AND = {
						has_policy_flag = pre_sapients_eat
						prev = {
							is_sapient = no
						}
					}
				}
				NOR = {
					OR = {
						has_valid_civic = civic_harm_hive
						has_valid_civic = civic_harm_machine
					}
				}
			}
		}
	}
	ai_will_do = {
		factor = 50
		modifier = {
			factor = 0
			NOR = { 
				has_trait = trait_delicious 
				from = { has_valid_civic = civic_hive_devouring_swarm }
				AND = {
					is_sapient = no
					from = {
						has_policy_flag = pre_sapients_eat
					}
				}
			}
		}
		modifier = {
			factor = 0
			from = {
				has_valid_ai_personality = yes
				NOR = {
					has_ai_personality_behaviour = purger
				}
			}
		}
	}
}
