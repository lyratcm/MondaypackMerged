
# Copy of the vanilla file concerning capitals

has_upgraded_capital = {
	custom_tooltip = {
		fail_text = "requires_building_capital"
		OR = {
			is_planet_class = pc_habitat
			has_modifier = resort_colony
			has_modifier = slave_colony
            owner = { is_regular_empire = no }
            owner = { is_master_computer = yes }			
			has_building = building_capital
			has_building = building_major_capital 
			has_building = building_system_capital
		}
	}
	custom_tooltip = {
		fail_text = "requires_building_machine_capital"
		OR = {
			is_planet_class = pc_habitat
			owner = { is_machine_empire = no }
			has_building = building_machine_capital
			has_building = building_machine_major_capital 
			has_building = building_machine_system_capital 			
		}
	}
	custom_tooltip = {
		fail_text = "requires_building_hive_capital"
		OR = {
			is_planet_class = pc_habitat
			owner = { is_hive_empire = no }
			has_building = building_hive_capital
			has_building = building_hive_major_capital
		}
	}
	custom_tooltip = {
		fail_text = "requires_building_hab_capital"
		OR = {	
			NOT = { is_planet_class = pc_habitat }
			has_building = building_hab_capital
			has_building = building_hab_major_capital
			has_building = building_master_computer_hab_capital_1
			has_building = building_master_computer_hab_capital_2
		}
	}
	custom_tooltip = {
		fail_text = "require_building_master_computer_capital"
		OR = {
			owner = { is_master_computer = no }
			is_planet_class = pc_habitat
			has_modifier = resort_colony
			has_modifier = slave_colony
			has_building = building_master_computer_capital_1
			has_building = building_master_computer_capital_2
			has_building = building_master_computer_capital_3
		}
	}
	custom_tooltip = {
		fail_text = "requires_building_resort_capital"
		OR = {	
			NOT = { has_modifier = resort_colony }
			has_building = building_resort_capital
			has_building = building_resort_major_capital
			has_building = building_master_computer_resort_capital_1
			has_building = building_master_computer_resort_capital_2
		}
	}		
	custom_tooltip = {
		fail_text = "requires_building_slave_capital"
		OR = {	
			NOT = { has_modifier = slave_colony }

		}
