##_[当前PC]	SP/MP通用
z_fw_trgr_cntr_core_pc_current = {
	OR = {
		AND = {#_SP
			is_multiplayer = no
			OR = {
				is_ai = no
				has_country_flag = z_fw_core_init_pc
			}
		}
		AND = {#_MP
			is_multiplayer = yes
			is_ai = no
		}
	}
}
