## SAME AS VANILLA RARE_1 BUT WON'T VANISH WHEN ZOOMED IN!

	star = {
		create_ambient_object = {
			type = "elven_nebula_rare_1"
			location = this
		}
		last_created_ambient_object = {
			set_location = {
				target = prev
				distance = 0
				angle = random
			}
		}
	}
}
