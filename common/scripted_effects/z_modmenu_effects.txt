	set_timed_global_flag = { flag = z_pf_flg_glbl_manual_checking_cd days = 7 }

	every_country = {
		limit = { has_country_flag = z_pr_flg_cntr_reg }
		z_pr_eft_cntr_mannual_check = yes
	}
	#_[默认]
	if = {
		limit = { has_global_flag = z_pf2_installed }
		every_country = {
			limit = { z_fw_trgr_cntr_core_playable_mod = yes }
			z_pf_eft_cntr_heup_manual = yes	#_[国内迁移]pf2
		}
	}
}
