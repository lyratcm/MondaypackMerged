cb_elven_celestial_tribute = {
	show_in_diplomacy = yes
	show_notification = yes
	potential = {
		is_country_type = default
		is_subject = no
		has_valid_civic = civic_elven_celestial_grace
		NOT = { is_in_federation_with = from }
	}
	is_valid = {
		from = {
			is_country_type = default
			is_neighbor_of = prev
		}
		custom_tooltip = {
			fail_text = elven_celestial_tribute_favor_fail
			success_text = elven_celestial_tribute_favor
			check_variable = {
				which = elven_celestial_favor_score
				value >= 5
			}
		}
		has_total_war_cb = no
		NOT = { is_overlord_to = FROM }
	}
}
