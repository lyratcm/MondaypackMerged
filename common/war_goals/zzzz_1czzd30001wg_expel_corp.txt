wg_expel_corp = {
    casus_belli = cb_expel_corp
    hide_if_no_cb = yes
    release_occupied_systems_on_status_quo = no
    allowed_peace_offers = {
        status_quo
        surrender
        demand_surrender
    }
    set_defender_wargoal = wg_megacorp_subjugate
    surrender_acceptance = -50
    potential = {
        has_total_war_cb = no
    }
    possible = {
        custom_tooltip = {
            fail_text = has_claimed_all_systems
            from = {
                is_country_type = default
                any_owned_planet = {
                    is_colony = yes
                    root = {
                        NOT = {
                            has_claim = prev.solar_system
                        }
                    }
                }
            }
        }
    }
    on_status_quo = {
        if = {
            limit = {
                from = {
                    is_criminal_syndicate = no
                }
            }
            every_owned_planet = {
                limit = {
                    has_branch_office = from
                }
                hidden_effect = {
                    branch_office_owner = {
                        add_resource = {
                            energy = 2000
                        }
                    }
                }
                close_branch_office = yes
            }
            custom_tooltip = mod_wg_expel_corp_status_quo_effects
        }
        else = {
            every_owned_planet = {
                limit = {
                    has_branch_office = from
                    planet_crime < 25
                }
                hidden_effect = {
                    branch_office_owner = {
                        add_resource = {
                            energy = 2000
                        }
                    }
                }
                close_branch_office = yes
            }
            every_owned_planet = {
                limit = {
                    has_branch_office = from
                }
        custom_tooltip = wg_expel_corp_status_quo_effects
                add_modifier = {
                    modifier = syndicate_crime_reduction
                    days = 3600
                }
            }
            custom_tooltip = mod_wg_expel_criminal_corp_status_quo_effects
        }
    }
    on_accept = {
        every_owned_planet = {
            limit = {
                has_branch_office = from
            }
            close_branch_office = yes
            hidden_effect = {
                owner = {
                    add_resource = {
                        energy = 1000
                    }
                }
            }
        }
        custom_tooltip = wg_expel_corp_effects
    }
    ai_weight = {
        weight = 10
    }
}
