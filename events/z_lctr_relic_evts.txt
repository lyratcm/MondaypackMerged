						}
					}
				}
				root = { z_fw_eft_cntr_misc_relic_site_counted = yes }	#_[挖掘地设置隔离debug]
				set_timed_star_flag = { flag = z_fw_flg_sstm_relic_site_counted days = 1 }	#_[挖掘地sstm隔离debug]
				set_timed_global_flag = { flag = z_lctr_flg_glbl_r days = 1 }	#_[挖掘地已设置]
				min_steps = 2
				max_steps = 17
				use_bypasses = yes
				random_system_planet = {
					limit = {
						has_owner = no
						is_colony = no
						has_anomaly = no
						NOR = {
							exists = archaeological_site
							has_planet_flag = hatchling_egg
						}
						OR = {
							is_planet_class = pc_molten
							is_planet_class = pc_barren
							is_planet_class = pc_barren_cold
							is_planet_class = pc_toxic
							is_planet_class = pc_frozen
							is_planet_class = pc_nuked
						}
					}
					set_name = "NAME_z_lctr_r_site_planet_0"	#_星球改名
					clear_deposits = yes
					set_planet_flag = z_lctr_flg_plnt_r
					create_archaeological_site = z_lctr_site
					every_country = {
						limit = { is_ai = no }
						begin_event_chain = {
							event_chain = z_lctr_chain_site
							target = this
						}
						#_通知
						country_event = { id = z_lctr_relic.10 days = 4 random = 3 }
						prev = {
							enable_special_project = {
								name = Z_LCTR_SP_SITE
								location = THIS
								owner = prev
							}
						}
					}
				}
			}
		}
	}
}
